func main() {

	var lstWords []string
	var wordLsts [][]string

	s := text //"Aligator dog cat cat aligator mouth cat dog dog cat dog aligator mouth horse aligator"
	s = strings.ToLower(s)
	// top 10
	newSlice := strings.Fields(s)
	sort.Sort(sort.StringSlice(newSlice))
	fmt.Println(newSlice)
	var word string
	for _, r := range newSlice {
		r = strings.Replace(r, ".", "", -1)
		if r == "-" {
			continue
		}
		if r != word {
			word = r
		} else {
			continue
		}
		lstWords = nil
		for _, r := range newSlice {
			r = strings.Replace(r, ".", "", -1)
			if r == word {
				lstWords = append(lstWords, r)
			}
		}
		wordLsts = append(wordLsts, lstWords)
	}
	// fmt.Println(wordLsts)

	// определить большинство
	sort.SliceStable(wordLsts, func(i, j int) bool {
		return len(wordLsts[i]) > len(wordLsts[j])
	})

	// fmt.Println(wordLsts)

	// вывести топ 10
	var top10Str []string
	top := 10
	//fmt.Println(wordLsts)
	for i, r := range wordLsts {
		if i == top {
			break
		}
		top10Str = append(top10Str, r[0])
	}
	fmt.Println(top10Str)
}
